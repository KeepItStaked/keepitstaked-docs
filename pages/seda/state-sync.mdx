 Create state-sync script:

```bash copy
RPC={{ rpc }}
LATEST_HEIGHT=$(curl -s $RPC/block | jq -r .result.block.header.height)
BLOCK_HEIGHT=$((LATEST_HEIGHT - 2000)); \
TRUST_HASH=$(curl -s "$RPC/block?height=$BLOCK_HEIGHT" | jq -r .result.block_id.hash)


sed -i \
  -e "s|^enable *=.*|enable = true|" \
  -e "s|^rpc_servers *=.*|rpc_servers = \"$RPC,$RPC\"|" \
  -e "s|^trust_height *=.*|trust_height = $BLOCK_HEIGHT|" \
  -e "s|^trust_hash *=.*|trust_hash = \"$TRUST_HASH\"|" \
  $HOME/{{ folder }}/config/config.toml

```
Apply state-sync configuration to the node
```bash copy
# stop and reset the node state
sudo systemctl stop {{ daemon }}
{{ daemon }} tendermint unsafe-reset-all --keep-addr-book
# run state-sync configuration
$HOME/state-sync.sh
# start the node 
sudo systemctl start {{ daemon }}
```
